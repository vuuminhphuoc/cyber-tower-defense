# v1.5 — AI Autoplay: Rule-Based Bot

## Goal

Create a deterministic AI bot that plays the game without any external API.
Useful for: testing, baseline AI, demo mode, fallback for LLM failures.

---

## Architecture

```
js/ai/
  ai-state.js       — serialize game state for AI
  ai-actions.js     — execute actions (place, upgrade, sell, collect, etc.)
  ai-rule-bot.js    — rule-based decision engine
  ai-controller.js  — tick loop + UI toggle
```

## AI Actions

### `AI.executeAction(action)`

Validates and executes an action.

```js
function executeAction(action) {
  switch (action.type) {
    case 'place_tower':
      return tryPlace(action.row, action.col, action.tower);
    case 'upgrade_tower':
      return upgradeTower(action.row, action.col);
    case 'sell_tower':
      return sellTower(action.row, action.col);
    case 'collect_all':
      return collectAllTokens();
    case 'send_wave':
      return startNextWave();
    case 'toggle_speed':
      return toggleSpeed();
    case 'wait':
      return true;
    default:
      return false;
  }
}
```

## Rule-Based Bot Logic

### Decision Priority

```
1. CRITICAL: Threat in final 2 cells with no wall → place wall
2. CRITICAL: No mower + threat in final 3 cells → emergency bomb/wall
3. HIGH: Lane has threat + no shooter → place cheapest shooter
4. HIGH: Token/coin available → collect
5. MEDIUM: Wave not started + board stable → send wave
6. MEDIUM: Enough credits + good placement → upgrade tower
7. MEDIUM: Water cell + no Proxy Node → place Proxy Node
8. LOW: Idle + credits > 100 → place producer
9. LOW: Idle + credits > 200 → place utility tower
10. DEFAULT: wait
```

### Rule Details

#### Rule 1: Emergency Wall

```js
function ruleEmergencyWall(state, actions) {
  for (const lane of state.lanes) {
    for (const threat of lane.threats) {
      if (threat.x > GRID_COLS * CELL_SIZE - 2 * CELL_SIZE) {
        // Threat near end
        const hasWall = lane.towers.some(t =>
          TOWER_TYPES[t].type === 'wall'
        );
        if (!hasWall) {
          const emptyCol = findEmptyCell(lane.row, 'right');
          if (emptyCol !== null) {
            actions.push({
              type: 'place_tower',
              tower: 'FIREWALL',
              row: lane.row,
              col: emptyCol
            });
          }
        }
      }
    }
  }
}
```

#### Rule 2: Lane Defense

```js
function ruleLaneDefense(state, actions) {
  for (const lane of state.lanes) {
    if (lane.threats.length > 0 && lane.towers.length === 0) {
      const emptyCol = findEmptyCell(lane.row, 'left');
      if (emptyCol !== null && state.credits >= 100) {
        actions.push({
          type: 'place_tower',
          tower: 'FIREWALL',
          row: lane.row,
          col: emptyCol
        });
      }
    }
  }
}
```

#### Rule 3: Economy Building

```js
function ruleEconomy(state, actions) {
  if (state.waveActive) return;
  if (state.credits < 50) return;

  // Find best producer cell
  const bestCell = findBestProducerCell(state);
  if (bestCell && state.credits >= 50) {
    actions.push({
      type: 'place_tower',
      tower: 'BITCOIN_MINER',
      row: bestCell.row,
      col: bestCell.col
    });
  }
}
```

#### Rule 4: Water Cell Proxy

```js
function ruleProxyNode(state, actions) {
  for (const lane of state.lanes) {
    for (const cell of lane.cells) {
      if (cell.type === 'water' && !cell.hasTower) {
        if (state.credits >= TOWER_TYPES['PROXY_NODE'].cost) {
          actions.push({
            type: 'place_tower',
            tower: 'PROXY_NODE',
            row: lane.row,
            col: cell.col
          });
        }
      }
    }
  }
}
```

#### Rule 5: Upgrade

```js
function ruleUpgrade(state, actions) {
  if (state.credits < 100) return;

  for (const lane of state.lanes) {
    for (const cell of lane.cells) {
      if (cell.hasTower && cell.towerKey !== 'PROXY_NODE') {
        const upgradeCost = TOWER_TYPES[cell.towerKey].cost * 0.6;
        if (state.credits >= upgradeCost) {
          actions.push({
            type: 'upgrade_tower',
            row: lane.row,
            col: cell.col,
            cost: upgradeCost
          });
        }
      }
    }
  }
}
```

## Controller

### `AI.start()`

```js
function startAI() {
  if (aiInterval) return;
  aiInterval = setInterval(() => {
    const state = getState();
    const actions = getLegalActions();
    const decisions = ruleBotDecide(state, actions);
    for (const action of decisions) {
      executeAction(action);
    }
  }, AI_TICK_RATE);
}
```

### `AI.stop()`

```js
function stopAI() {
  if (aiInterval) {
    clearInterval(aiInterval);
    aiInterval = null;
  }
}
```
