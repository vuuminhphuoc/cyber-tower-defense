# v2.0 — AI Autoplay via User's LLM Provider

## Goal

User enters their API credentials (provider, key, URL, model).
AI reads game state, decides actions, plays autonomously.

---

## Architecture

```
js/ai/
  ai-state.js           — game state serialization (from v1.5)
  ai-actions.js         — action execution (from v1.5)
  ai-rule-bot.js        — rule-based fallback (from v1.5)
  ai-controller.js      — tick loop + mode switching (from v1.5)
  ai-provider.js        — LLM API adapter
  ai-prompt.js          — system prompt + message builder
  ai-settings.js        — settings UI + localStorage persistence
  ai-validator.js       — validate LLM output against legal actions
```

## User Settings

### Fields

| Field | Default | Description |
|---|---|---|
| Provider | `openai-compatible` | LLM provider type |
| API URL | `https://api.openai.com/v1/chat/completions` | Endpoint |
| API Key | (empty) | User's secret key |
| Model | `gpt-4o-mini` | Model name |
| Temperature | `0.2` | Low = more deterministic |
| Max Tokens | `500` | Response limit |
| Tick Interval | `2000` | ms between AI decisions |
| Max Actions | `2` | Actions per decision |
| Play Style | `safe` | safe / aggressive / economy / speedrun |
| Auto-fallback | `true` | Fall back to rule bot on error |

### Storage

```js
// localStorage key
const AI_SETTINGS_KEY = 'cyber_defenders_ai_settings';

// Save
function saveAISettings(settings) {
  localStorage.setItem(AI_SETTINGS_KEY, JSON.stringify(settings));
}

// Load
function loadAISettings() {
  try {
    return JSON.parse(localStorage.getItem(AI_SETTINGS_KEY)) || defaultSettings();
  } catch {
    return defaultSettings();
  }
}
```

### Security

- API key stored in `localStorage` only
- Never sent to any server except the user's chosen provider
- Clear key button available
- Warning: "Do not use on shared/public computers"

## Provider Adapters

### Supported Providers

| Provider | API Style | Auth Header |
|---|---|---|
| OpenAI | `/v1/chat/completions` | `Authorization: Bearer {key}` |
| OpenRouter | OpenAI-compatible | `Authorization: Bearer {key}` |
| Groq | OpenAI-compatible | `Authorization: Bearer {key}` |
| Together AI | OpenAI-compatible | `Authorization: Bearer {key}` |
| DeepSeek | OpenAI-compatible | `Authorization: Bearer {key}` |
| Anthropic | `/v1/messages` | `x-api-key: {key}` |
| Google Gemini | `/v1/models/{model}:generateContent` | `x-goog-api-key: {key}` |
| Custom | User-defined | User-defined |

### Adapter Interface

```js
const AIProvider = {
  async call(config, messages) {
    switch (config.provider) {
      case 'openai-compatible':
        return callOpenAICompatible(config, messages);
      case 'anthropic':
        return callAnthropic(config, messages);
      case 'gemini':
        return callGemini(config, messages);
      default:
        throw new Error(`Unknown provider: ${config.provider}`);
    }
  }
};
```

### OpenAI-Compatible Adapter

```js
async function callOpenAICompatible(config, messages) {
  const headers = {
    'Content-Type': 'application/json'
  };

  if (config.apiKey) {
    headers['Authorization'] = `Bearer ${config.apiKey}`;
  }

  const res = await fetch(config.apiUrl, {
    method: 'POST',
    headers,
    body: JSON.stringify({
      model: config.model,
      messages,
      temperature: config.temperature,
      max_tokens: config.maxTokens
    })
  });

  if (!res.ok) {
    const err = await res.text();
    throw new Error(`API error ${res.status}: ${err}`);
  }

  const data = await res.json();
  return data.choices?.[0]?.message?.content || '';
}
```

### Anthropic Adapter

```js
async function callAnthropic(config, messages) {
  const systemMsg = messages.find(m => m.role === 'system');
  const userMsg = messages.filter(m => m.role !== 'system');

  const res = await fetch(config.apiUrl, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': config.apiKey,
      'anthropic-version': '2023-06-01'
    },
    body: JSON.stringify({
      model: config.model,
      system: systemMsg?.content || '',
      messages: userMsg,
      max_tokens: config.maxTokens,
      temperature: config.temperature
    })
  });

  if (!res.ok) {
    const err = await res.text();
    throw new Error(`API error ${res.status}: ${err}`);
  }

  const data = await res.json();
  return data.content?.[0]?.text || '';
}
```

### Gemini Adapter

```js
async function callGemini(config, messages) {
  const systemMsg = messages.find(m => m.role === 'system');
  const userMsg = messages.filter(m => m.role !== 'system');

  const contents = userMsg.map(m => ({
    role: m.role === 'assistant' ? 'model' : 'user',
    parts: [{ text: m.content }]
  }));

  const res = await fetch(
    `${config.apiUrl}/v1/models/${config.model}:generateContent?key=${config.apiKey}`,
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        contents,
        systemInstruction: systemMsg ? { parts: [{ text: systemMsg.content }] } : undefined,
        generationConfig: {
          temperature: config.temperature,
          maxOutputTokens: config.maxTokens
        }
      })
    }
  );

  if (!res.ok) {
    const err = await res.text();
    throw new Error(`API error ${res.status}: ${err}`);
  }

  const data = await res.json();
  return data.candidates?.[0]?.content?.parts?.[0]?.text || '';
}
```

## Response Validator

```js
function validateAIResponse(response, legalActions) {
  try {
    const parsed = JSON.parse(response);
    if (!parsed.actions || !Array.isArray(parsed.actions)) {
      return { valid: false, error: 'Missing actions array' };
    }

    const validActions = [];
    for (const action of parsed.actions) {
      const match = legalActions.find(la =>
        JSON.stringify(la) === JSON.stringify(action)
      );
      if (match) {
        validActions.push(match);
      }
    }

    return {
      valid: true,
      thought: parsed.thought || '',
      actions: validActions
    };
  } catch (e) {
    return { valid: false, error: `JSON parse error: ${e.message}` };
  }
}
```

## AI Tick Loop

```js
async function aiTick() {
  if (!aiConfig.enabled || aiConfig.mode !== 'llm') return;
  if (aiBusy) return; // prevent overlapping requests

  aiBusy = true;
  try {
    const state = getState();
    const legalActions = getLegalActions();

    if (legalActions.length === 0) {
      aiBusy = false;
      return;
    }

    const messages = [
      { role: 'system', content: buildSystemPrompt(aiConfig) },
      { role: 'user', content: buildUserPrompt(state, legalActions, aiConfig.playStyle) }
    ];

    const response = await AIProvider.call(aiConfig, messages);
    const result = validateAIResponse(response, legalActions);

    if (!result.valid) {
      console.warn('AI invalid response:', result.error);
      if (aiConfig.autoFallback) {
        // Use rule bot for this tick
        const ruleActions = ruleBotDecide(state, legalActions);
        for (const action of ruleActions.slice(0, aiConfig.maxActions)) {
          executeAction(action);
        }
      }
      return;
    }

    // Execute validated actions
    for (const action of result.actions.slice(0, aiConfig.maxActions)) {
      executeAction(action);
    }

    // Update status UI
    updateAIStatus(result.thought, result.actions.length);

  } catch (e) {
    console.error('AI error:', e);
    updateAIStatus(`Error: ${e.message}`, 0);

    if (aiConfig.autoFallback) {
      const state = getState();
      const legalActions = getLegalActions();
      const ruleActions = ruleBotDecide(state, legalActions);
      for (const action of ruleActions.slice(0, aiConfig.maxActions)) {
        executeAction(action);
      }
    }
  } finally {
    aiBusy = false;
  }
}
```

## Error Handling

| Error | Action |
|---|---|
| Network error | Retry once, then fallback to rule bot |
| 429 Rate limit | Exponential backoff: 2s → 5s → 10s |
| 401 Unauthorized | Show "Invalid API key" message |
| 402 Payment required | Show "Insufficient credits" message |
| JSON parse error | Fallback to rule bot |
| Invalid actions | Execute only valid subset |
| Timeout (10s) | Cancel request, fallback to rule bot |

## Rate Limiting

```js
const rateLimiter = {
  lastRequest: 0,
  minInterval: 1000,
  backoff: 1000,
  maxBackoff: 10000,

  async wait() {
    const now = Date.now();
    const elapsed = now - this.lastRequest;
    if (elapsed < this.minInterval) {
      await sleep(this.minInterval - elapsed);
    }
    this.lastRequest = Date.now();
  },

  onError() {
    this.backoff = Math.min(this.backoff * 2, this.maxBackoff);
    return this.backoff;
  },

  onSuccess() {
    this.backoff = 1000;
  }
};
```

## Cost Tracking

```js
const costTracker = {
  requests: 0,
  totalTokens: 0,
  lastDecision: null,

  track(response, tokenCount) {
    this.requests++;
    this.totalTokens += tokenCount;
    this.lastDecision = {
      time: Date.now(),
      thought: response.thought,
      actions: response.actions.length
    };
  },

  getStats() {
    return {
      requests: this.requests,
      tokens: this.totalTokens,
      lastDecision: this.lastDecision
    };
  }
};
```
